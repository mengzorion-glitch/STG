        centerX - barWidth / 2 + 5,
        centerY + 5,

        barHeight - 10

      loadingText.destroy();
      progressBar.destroy();
      progressBox.destroy();
    });

      dy = this.touchTarget.y - centerY;

      // æ­£è¦åŒ–
      const length = Math.sqrt(dx * dx + dy * dy);
      if (length > 10) {

        dy /= length;
      } else {
        dx = 0;
        dy = 0;
      }

    // å°è§’ç·šæ­£è¦åŒ–

      dx *= normalizer;
      dy *= normalizer;

  }
